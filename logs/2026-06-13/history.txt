def call_groq_api_safe(prompt: str) -> str:
    """KeyError 안전처리 버전"""
    import requests
    headers = {
        "Content-Type": "application/json",
        "Authorization": f"Bearer {os.getenv('GROQ_API_KEY')}"
    }
    body = {
        "model": "llama-3.3-70b-versatile",
        "max_tokens": 1500,
        "messages": [{"role": "user", "content": prompt}]

            headers=headers, json=body, timeout=30
        )
        data = response.json()
        if "choices" not in data:
            # API 에러 응답 형식 (예: {"error": {"message": "..."}})
            error_msg = data.get("error", {}).get("message", "알 수 없는 오류")
            raise KeyError(f"API 응답에 choices 없음: {error_msg}")
        return data["choices"][0]["message"]["content"]
    except KeyError as e:
        print(f"❌ API 응답 형식 오류: {e}")
        return "(요약 실패 - API 응답 형식 오류)"
    except requests.exceptions.RequestException as e:
        print(f"❌ 네트워크 오류: {e}")
