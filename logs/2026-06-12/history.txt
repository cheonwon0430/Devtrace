cat > multi.py << 'EOF'
import requests
import time
import threading

URLS = [
    'https://example.com',
    'https://www.python.org',
    'https://www.github.com',
    'https://www.stackoverflow.com',
    'https://www.wikipedia.org',
    'https://www.naver.com',
    'https://www.daum.net',
    'https://www.yahoo.com',
    'https://www.google.com',
    'https://www.sejong.ac.kr',
]

def fetch(url):
    t0 = time.time()
    try:
        r = requests.get(url, timeout=10)
        print(f"{url} -> {r.status_code} {r.reason} ({time.time()-t0:.3f}s)")
    except Exception as e:
        print(f"{url} -> Error ({e})")

start = time.time()
threads = [threading.Thread(target=fetch, args=(u,)) for u in URLS]
for t in threads:
    t.start()
for t in threads:
    t.join()
print(f"\nTotal time: {time.time()-start:.2f}s")
EOF
